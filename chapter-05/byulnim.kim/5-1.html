예제 32 : 푸시/팝

스택에 데이터를 저자하는 push 함수와 데이터를 추출하는 pop 함수를 작성한다.

프로그램 : 스택 (stack)

            var _count = 0;
            var _stack = [];
            function push(val) {
              document.getElementById("pushVal").value = "";
    
              _stack[_count] = val;
              _count++;
    
              drawStack();
            }
            function pop() {
              var stackLength = _stack.length;
              _stack.splice([stackLength-1], 1);
              _count--;
    
              drawStack();
            }
            function drawStack() {
              var stackHtml = "";
              for (i=_stack.length-1; i<=0; i--) {
                stackHtml += _stack[i] + "
";
              }
              document.getElementById("stack").innerHTML=stackHtml;
            }
