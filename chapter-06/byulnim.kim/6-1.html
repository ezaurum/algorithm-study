            function strcmp ( str1, str2 ) {
              return ( ( str1 == str2 ) ? 0 : ( ( str1 > str2 ) ? 1 : -1 ) );
            }
      
            var tnode = [
              {left: 1, name: 'Matilda', right: 2},
              {left: 3, name: 'Candy', right: 4},
              {left: 5, name: 'Rolla', right: null},
              {left: null, name: 'Ann', right: null},
              {left: 6, name: 'Emy', right: 7},
              {left: null, name: 'Nancy', right: null},
              {left: null, name: 'Eluza', right: null},
              {left: null, name: 'Lisa', right: null},
            ];
      
            function searchByKey(k) {
              var p = 0;
              while(p != null) {
                if (strcmp(k, tnode[p].name) === 0) {
                  document.getElementById('result').innerText = "찾았습니다";
                  break;
                } 
                else if (strcmp(k, tnode[p].name) < 0) {
                  p = tnode[p].left;
                  document.getElementById('result').innerText = "없는 이름입니다.";
                } else {
                  p = tnode[p].right;
                  document.getElementById('result').innerText = "없는 이름입니다.";
                }
              }
            }
