예제 2 : 도수분포(히스토그램)

0~100 사이의 정수를 10점 간격으로 나누어 (0~9, 10~10, ..., 90~99, 100의 11등급) 각 등급의 도수분포(히스토그램)를 구하라.

프로그램 : 도수분포(히스토그램)

        var a = [35, 25, 56, 78, 43, 66, 71, 73, 80, 90, 0, 73, 35, 65, 100, 78, 80, 85, 35, 50];
        var rank;
        var histo = new Array;
        function main() {
          for (i=0; i<11; i++) {
            histo.push(new Array);
          }
          var cnt = 0;
          for (i=0; i<a.length; i++) {
            rank = (a[i]-(a[i]%10))/10;
            histo[rank].push(cnt++);
          }
          for (i=0; i<histo.length; i++) {
            console.log(i + "등급 - : " + histo[i].length);
          }
        }
        main();
