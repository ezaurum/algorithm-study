1-3. 순위 매김

예제 3 : 간단한 순위 매기기 방법

시험 점수의 점수별 순위를 구한다.

프로그램 : 순위 매김

          var a = [56, 25, 67, 88, 100, 61, 55, 67, 76, 56];
          var rank = new Array;
          
          function main() {
            for (i=0; i<a.length; i++) {
              rank.push(1);
              for (j=0; j<a.length; j++) {
                if (a[j] > a[i]) {
                  rank[i]++;
                }
              }
              console.log(a[i] + "점은 a 목록중에 " + rank[i] + "등입니다.");
            }
          }
          main();

결과
