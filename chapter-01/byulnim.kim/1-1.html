예제 1 : n개 중에 r개를 선택하는 조합 nCr을 구하라.

점화식 (nCr 계산)

        function main() {
            var n, r;
            for (n=0; n <= 5; n++) {
                var row = "";
                for (r=0; r <= n; r++) {
                    row += n + " C " + r +"=" + combi(n, r) + " ";
                }
                console.log(row);
            }
        }

        function combi(n, r) {
            var i;
            var p= 1;
            for (i=1; i<=r; i++) {
                p = p*(n-i+1)/i;
            }
            return p;
        }

        main();
